void printMatrix(int** matrix, int N)
{
    int i,j;
    for (i=0; i<N; i++)
    {
        for(j=0; j<N; j++)
        {
            printf("%5d", matrix[i][j]);
        }   
    printf("\n");
    }
}

void printVector(int* v, int N)
{
    int i;

    }
    printf("}\n\n");
}

void resetVector(int* v, int N)
{
    int i;
    for (i=0; i<N; i++)
    {
        v[i] = 0;
    }
}

void sequentialBlockAlgorithmJOSEP(int* V, int** M, int N, int B, int x)
{
	int i, j, m,k;
	m = 0; k = 0;
	for (i=0; i<N; i=i+B)
	{
		for (j=i; j<N; j=j+B)
		{
			for (m=i; m<(j+B); m++)
			{
				if(i==j)
				{
					for (k=m; k<(j+B); k++)
					{
						V[m]=V[m] + x * M[m][k];						
					}
				}
				else
				{
					for (k=j; k<(j+B); k++)
					{
						V[m]=V[m] + x * M[m][k];
					}			
				}		
			}
			printf("%5d", x * M[m][k]);
            Sleep(250); // DELAY TO HUMANLY DISPLAY MATRIX FOLLOW PATH		
		}
		printf("\n");
	}
}
